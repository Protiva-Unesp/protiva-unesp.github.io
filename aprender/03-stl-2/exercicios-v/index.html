Exercício C - a-Good String

Objetivo -> Contar quantas modificações precisam ser feitas para transformar a string dada em uma a-Good String. (É garantido que o tamanho da string é uma potência de 2) O que é uma a-Good String ? aaaabbcd aaaacdbb bbcdaaaa cdbbaaaa

Como fazer essas strings? O problema fica mais fácil se eu dividir a string ao meio ? Ex: aaaabbcd -> aaaa bbcd A string bbcd seria uma b-Good_String Isto é, podemos dividir a string bbcd ao meio também :)

ll merge(string &s, char a, ll ini, ll fim){
    if(ini == fim){
        if(s[ini] != a)return 1;
            else return 0;
    }
    ll meio = (ini + fim)/2;
    ll res = 1e17;
    ll changes = 0;
    for(int i = ini; i <= meio; i++){
        if(s[i] != a)changes++;
    }
    changes += merge(s, a+1, meio+1, fim);
    res = min(res, changes);
    changes = 0;
    for(int i = meio+1; i <= fim; i++){
        if(s[i] != a) changes++;
    }
    changes += merge(s, a+1, ini, meio);
    res = min(res, changes);
    return res;
}

int main(){
    ll n;

    ll i = 1;
    ll cont = 0;
    while(i <= n){
        i*= 2;
        cont++;
    }
    ll fim = ( 1LL << cont) - 1LL;

ll func(ll pos, ll ini, ll fim, ll val){
    if(ini > r || fim < l || val == 0)return 0;
    if(val == 1){
        if(pos < l || pos > r) return 0;
            return 1;
    }
    ll pos1 = (pos+ini-1LL)/2LL;
    ll pos2 = (fim+pos+1LL)/2LL;
    ll soma = 0;
    if(val&1 && pos >= l && pos <= r)soma++;
    soma += func(pos1, ini, pos-1LL, val/2LL)
    + func(pos2, pos+1LL, fim, val/2LL);
    return soma;
}
